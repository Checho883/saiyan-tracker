```python
class GameEvent(BaseModel):
    type: str  # 'streak_milestone', 'level_up', 'transformation'

class StreakMilestoneEvent(GameEvent):
    type: str = "streak_milestone"
    tier: int  # The milestone number (3, 7, 21...)
    streak: int  # Current streak count
    badge_name: str  # DBZ-themed name
    scope: str  # "overall" or "habit"

class LevelUpEvent(GameEvent):
    type: str = "level_up"
    attribute: str
    old_level: int
    new_level: int
    title: str | None = None

class TransformationEvent(GameEvent):
    type: str = "transformation"
```

```python
events = []
if is_checking:
    # Overall streak milestones
    from app.services.achievement_service import (
        detect_streak_milestones, detect_attribute_level_change,
        record_achievement, get_milestone_badge_name
    )
    overall_milestones = detect_streak_milestones(
        streak_result["current_streak"] - 1 if streak_result["current_streak"] > 0 else 0,
        streak_result["current_streak"]
    )
    # NOTE: For overall streak, old_streak = current - 1 because update_overall_streak already incremented.
    # For habit streak same pattern applies.
    habit_milestones = detect_streak_milestones(
        habit_streak_result["current_streak"] - 1 if habit_streak_result["current_streak"] > 0 else 0,
        habit_streak_result["current_streak"]
    )
    for m in overall_milestones:
        record_achievement(db, user_id, "streak_milestone", f"overall_streak_{m}",
                          milestone_type="streak", metadata={"streak": m, "scope": "overall"})
        events.append({"type": "streak_milestone", "tier": m,
                       "streak": streak_result["current_streak"],
                       "badge_name": get_milestone_badge_name(m), "scope": "overall"})
    for m in habit_milestones:
        record_achievement(db, user_id, "streak_milestone", f"habit_{habit_id}_streak_{m}",
                          milestone_type="streak", metadata={"streak": m, "scope": "habit", "habit_id": str(habit_id)})
        events.append({"type": "streak_milestone", "tier": m,
                       "streak": habit_streak_result["current_streak"],
                       "badge_name": get_milestone_badge_name(m), "scope": "habit"})
    # Attribute level-up
    new_attr_xp = getattr(user, f"{habit.attribute}_xp")
    level_change = detect_attribute_level_change(old_attr_xp, new_attr_xp, habit.attribute)
    if level_change:
        record_achievement(db, user_id, "attribute_level_up",
                          f"{habit.attribute}_{level_change['new_level']}",
                          metadata=level_change)
        events.append({"type": "level_up", **level_change, "attribute": habit.attribute})
```
