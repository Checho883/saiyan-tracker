   if (result.events && result.events.length > 0) {
     for (const event of result.events) {
       if (event.type === 'level_up') {
         ui.enqueueAnimation({
           type: 'level_up',
           attribute: event.attribute,
           oldLevel: event.old_level,
           newLevel: event.new_level,
           title: event.title ?? null,
         });
       }
       if (event.type === 'streak_milestone') {
         ui.enqueueAnimation({
           type: 'streak_milestone',
           tier: event.tier,
           streak: event.streak,
           scope: event.scope ?? 'overall',
           badgeName: event.badge_name ?? `Milestone ${event.tier}`,
         });
       }
     }
   }

   // Zenkai recovery: fires when zenkai_activated AND is_perfect_day
   if (result.zenkai_activated && result.is_perfect_day && result.is_checking) {

5. **frontend/src/store/statusStore.ts** — Create new Zustand store:
   ```typescript
   import { create } from 'zustand';
   import type { StatusResponse } from '../types';
   import { statusApi } from '../services/api';

   interface StatusState {
     status: StatusResponse | null;
     isLoaded: boolean;
     fetchStatus: (date: string) => Promise<void>;
     clearStatus: () => void;
   }

   export const useStatusStore = create<StatusState>((set) => ({
     status: null,
     isLoaded: false,
     fetchStatus: async (date) => {
       try {
         const status = await statusApi.get(date);
         set({ status, isLoaded: true });
       } catch {
         set({ status: null, isLoaded: true });
       }
     },
     clearStatus: () => set({ status: null, isLoaded: false }),
   }));
   ```
